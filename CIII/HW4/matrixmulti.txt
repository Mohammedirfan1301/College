Matrix operator * (const Matrix& one, const Matrix& two)
{
    Matrix temp;

    for(int i = 0; i < 3; i++)
    {
        for(int j = 0; j < 3; j++)
        {
            for (int k = 0; k < 3; k++)
            {
                   temp.e[i][j] += one.e[i][k] * two.e[k][j];
            }
        }

    }

    return temp;
}
